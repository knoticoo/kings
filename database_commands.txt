## Database Commands for King's Choice

# Make sure you're in the right directory
cd /workspace

# Basic player count
sqlite3 kings_choice.db "SELECT COUNT(*) as 'Total Players' FROM players;"

# List all players with details
sqlite3 kings_choice.db -header -column "
SELECT 
    id,
    name as 'Player Name',
    CASE WHEN is_current_mvp = 1 THEN 'YES' ELSE 'NO' END as 'Current MVP',
    mvp_count as 'MVP Count',
    CASE WHEN is_excluded = 1 THEN 'YES' ELSE 'NO' END as 'Excluded'
FROM players 
ORDER BY name;
"

# Check all tables
sqlite3 kings_choice.db ".tables"

# Interactive mode
sqlite3 kings_choice.db

# Alternative: Use Python (no sqlite3 installation needed)
python3 -c "
import sqlite3
conn = sqlite3.connect('kings_choice.db')
cursor = conn.cursor()
cursor.execute('SELECT COUNT(*) FROM players;')
print(f'Total players: {cursor.fetchone()[0]}')
cursor.execute('SELECT name, is_current_mvp, mvp_count FROM players ORDER BY name;')
players = cursor.fetchall()
if players:
    print('Players:')
    for name, is_mvp, count in players:
        mvp = ' (Current MVP)' if is_mvp else ''
        print(f'  - {name} (MVP count: {count}){mvp}')
else:
    print('No players found.')
conn.close()
"
